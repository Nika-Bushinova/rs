Write a deepCopy function to deep copy the value passed to it. The function must take a number, string, boolean, null, undefined, hash or array and return a copy of it, including all sub-hashes, sub-arrays, etc.

 
                  function deepCopy(data) {
                     let dataCopy;
                     if (typeof data !== 'object' || !data) {
                        dataCopy = data
                     } else {
                        if (Array.isArray(data)) {
                           dataCopy = Object.assign([], data)
                           for (let i = 0; i < data.length; i++) {
                              dataCopy[i] = deepCopy(data[i])
                           }
                        } else {
                           dataCopy = Object.assign({}, data)
                           for (let key in data) {
                              dataCopy[key] = deepCopy(data[key])
                           }
                        }
                     } return dataCopy
                  }
